			) shi_max on (shi_max.order_id=shi.order_id and shi_max.id=shi.id)

,sku_metro Sku,nombre_metro Material,marca_metro Marca,link_producto,url_imagen_metro  imagen_vtex,to_char(to_date(fecha_scraping_metro ,'dd-mm-yyyy'),'yyyymmdd')Fecha
,b.id_ean_principal, max(cast(precio_regular_metro as numeric(14,2))) precio_regular , max(cast(precio_online_metro as numeric(14,2))) precio_online
into #sku_metro from #scraping_metro a
left join dl_sp_capa_semantica.dw_cencofcic_dim_material b on right('000000000000000000'+ a.sku_metro ,18)=b.id_material
group by split_part(categoria_metro,'/',2) ,split_part(categoria_metro,'/',3) ,split_part(categoria_metro,'/',4) 
,sku_metro ,nombre_metro ,marca_metro ,link_producto,url_imagen_metro ,to_char(to_date(fecha_scraping_metro ,'dd-mm-yyyy'),'yyyymmdd'),b.id_ean_principal
;
