-- DROP PROCEDURE dl_sp_dashboards_ecommerce.sp_venta_online_metro_almacenes();

CREATE OR REPLACE PROCEDURE dl_sp_dashboards_ecommerce.sp_venta_online_metro_almacenes()
	LANGUAGE plpgsql
AS $$
	
	
	
	

begin


---METRO ALMACENES

drop table if exists #fecha_dims_Timestamp;
WITH tmp_variables AS (
SELECT 
cast(to_char(dateadd(day,-10,convert_timezone('GMT','GMT+5',getdate())),'yyyymmdd') as int)AS var_fecha_ini,
cast(to_char(dateadd(day,-1,convert_timezone('GMT','GMT+5',getdate())),'yyyymmdd') as int)AS var_fecha_fin
)


select * into #fecha_dims_Timestamp  from  tmp_variables;

drop table if exists #FECHA;
select SK_DIA,SK_SEMANA_ANIO,SK_SEMANA,DESC_ABRV_DIA_SEM,DESC_ABRV_DIA_SEM DIA into #FECHA from dl_sp_maestras.dim_dia
where SK_DIA>=(select var_fecha_ini from #fecha_dims_Timestamp ) and SK_DIA<=(select var_fecha_fin from #fecha_dims_Timestamp );

drop table if exists #ticket;
SELECT DISTINCT codfecha,codtiendalost COD_TIENDA,hora,caja,ticket,cajera,tipocomprobante,numfolio,codticket, 
codtiendalost+'-'+CAST(a.codfecha as CHAR(8))+ '-'+ Caja+'-'+ Ticket+'-'+Left(Hora,2)+RIGHT(Hora,2) as CodTicket_construido
INTO #ticket FROM per_super_pd_int.ticket a
inner join #FECHA d on a.codfecha =d.SK_DIA
WHERE  CodTiendaLost IN ('S038','S014','H010','S043','S085','S010','S029','S089','S044','S055','S053','S046','S051','S076','H011','S036','S092','S025','S034');

drop table if exists #piso;
SELECT DISTINCT COD_CLIENTE,COD_TICKET,COD_DIA,A.COD_MATERIAL,MTO_VENTA_NETA,MTO_VENTA_BRUTA,CNT_VENTA,COD_CADENA,COD_TIENDA,COD_CANAL,MTO_MARGEN
INTO #piso
FROM per_super_pd_stg.TRX_TICKET a
INNER JOIN #FECHA B ON A.COD_DIA=B.SK_DIA
WHERE COD_TIENDA IN ('S038','S014','H010','S043','S085','S010','S029','S089','S044','S053','S055','S046','S051','S076','H011','S036','S092','S025','S034') AND COD_CANAL=2;


drop table if exists #MATERIALES_UNIDO;

SELECT ID_MATERIAL,ID_EAN_PRINCIPAL,A.SK_DIA_CREACION_MATERIAL,DESC_MATERIAL,DESC_MARCA,ID_UM_BASE,ID_PROVEEDOR,DESC_PROVEEDOR,
ID_GRUPO_ARTICULO,DESC_GRUPO_ARTICULO,ID_CATEGORIA,DESC_CATEGORIA,DESC_JEFE_CATEGORIA,DESC_JEFE_CATEGORIA_MP,DESC_JEFE_CATEGORIA_IMP,ID_SUB_CATEGORIA,DESC_SUB_CATEGORIA,ID_SECCION,DESC_SECCION,ID_DEPARTAMENTO,DESC_DEPARTAMENTO,ID_AREA,DESC_AREA,ID_GERENCIA_COMERCIAL,DESC_GERENCIA_COMERCIAL,ID_GERENTE,DESC_GERENTE,
DESC_UNIDAD_NEGOCIO,E.SK_CATEGORIA,A.SK_MATERIAL,B.SK_MARCA,ID_TIPO_IGV
INTO #MATERIALES_UNIDO FROM dl_sp_capa_semantica.dw_cencofcic_dim_material A
INNER JOIN dl_sp_capa_semantica.dw_cencofcic_dim_marca B ON A.SK_MARCA=B.SK_MARCA
INNER JOIN dl_sp_capa_semantica.dw_cencofcic_dim_proveedor C ON A.SK_PROVEEDOR=C.SK_PROVEEDOR
INNER JOIN dl_sp_maestras.dim_grupo_articulo D ON A.SK_GRUPO_ARTICULO=D.SK_GRUPO_ARTICULO
INNER JOIN dl_sp_maestras.dim_categoria E ON D.SK_CATEGORIA=E.SK_CATEGORIA
INNER JOIN dl_sp_maestras.dim_jefe_categoria F ON D.SK_JEFE_CATEGORIA=F.SK_JEFE_CATEGORIA
INNER JOIN dl_sp_maestras.dim_jefe_categoria_mp G ON D.SK_JEFE_CATEGORIA_MP=G.SK_JEFE_CATEGORIA_MP
INNER JOIN dl_sp_maestras.dim_jefe_categoria_imp H ON D.SK_JEFE_CATEGORIA_IMP=H.SK_JEFE_CATEGORIA_IMP
INNER JOIN dl_sp_maestras.dim_sub_categoria I ON D.SK_SUB_CATEGORIA=I.SK_SUB_CATEGORIA
INNER JOIN dl_sp_maestras.dim_seccion J ON D.SK_SECCION=J.SK_SECCION
INNER join dl_sp_maestras.dim_departamento K ON D.SK_DEPARTAMENTO=K.SK_DEPARTAMENTO
INNER JOIN dl_sp_maestras.dim_area L ON D.SK_AREA=L.SK_AREA
INNER JOIN dl_sp_maestras.dim_gerencia_comercial M ON D.SK_GERENCIA_COMERCIAL=M.SK_GERENCIA_COMERCIAL
INNER JOIN dl_sp_maestras.dim_gerente N ON D.SK_GERENTE=N.SK_GERENTE;

drop table if exists #MATERIAL;
SELECT id_material as "COD_MATERIAL",ID_EAN_PRINCIPAL as "EAN",SK_DIA_CREACION_MATERIAL as "FECHA_CREACION",DESC_MATERIAL,DESC_MARCA as "MARCA"
,ID_UM_BASE as "UMB",ID_PROVEEDOR as "COD_PROVEEDOR",DESC_PROVEEDOR,ID_GRUPO_ARTICULO as "COD_GA",DESC_GRUPO_ARTICULO as"GRUPO_ARTICULO",
ID_CATEGORIA as "COD_CATEGORIA",DESC_CATEGORIA as "CATEGORIA",DESC_JEFE_CATEGORIA"JEFE_CATEGORIA"
,DESC_JEFE_CATEGORIA_MP"JEFE_CATEGORIA_MP",DESC_JEFE_CATEGORIA_IMP"JEFE_CATEGORIA_IMP",
ID_SUB_CATEGORIA"COD_SUB_CATEGORIA",DESC_SUB_CATEGORIA"SUB_CATEGORIA",ID_SECCION"COD_SECCION",DESC_SECCION"SECCION",
ID_DEPARTAMENTO"COD_DEPARTAMENTO",DESC_DEPARTAMENTO"DEPARTAMENTO",ID_AREA"COD_AREA",DESC_AREA"AREA",ID_GERENCIA_COMERCIAL"COD_GERENCIA",
DESC_GERENCIA_COMERCIAL"GERENCIA_COMERCIAL",ID_GERENTE"COD_GERENTE", DESC_GERENTE"GERENTE",DESC_UNIDAD_NEGOCIO UNIDAD_NEGOCIO,SK_CATEGORIA,SK_MATERIAL,SK_MARCA,ID_TIPO_IGV
INTO #MATERIAL FROM #MATERIALES_UNIDO;


drop table if exists  #PEDIDO_MAT;
SELECT E.*,M.DESC_MATERIAL,M.MARCA,M.GRUPO_ARTICULO,M.CATEGORIA,GERENTE,M.UNIDAD_NEGOCIO ,M.SECCION,M.DEPARTAMENTO,M.GERENCIA_COMERCIAL
INTO #PEDIDO_MAT  FROM #piso E
INNER JOIN #MATERIAL M
ON E.COD_MATERIAL=M.COD_MATERIAL;

alter table #PEDIDO_MAT add NUM_ORDEN_ECOMMERCE varchar(50);

drop table if exists #TKT_WEB;
SELECT DISTINCT COD_DIA,COD_TIENDA,COD_TICKET,NUM_ORDEN_ECOMMERCE,COD_CLIENTE,SUBSTRING(NUM_ORDEN_ECOMMERCE,1,8) SEQUENCE_ONLINE,COD_CANAL,COD_CAJA,NUM_TICKET
INTO #TKT_WEB
FROM per_super_pd_stg.TRX_TICKET_CAB a
INNER JOIN #FECHA B ON A.COD_DIA=B.SK_DIA
WHERE COD_TIENDA IN ('S038','S014','H010','S043','S085','S010','S029','S089','S044','S053','S055','S046','S051','S076','H011','S036','S092','S025','S034') AND COD_CANAL=2;

alter table #PEDIDO_MAT add TipoComprobante varchar(50);
alter table #PEDIDO_MAT add NumFolio varchar(50);

UPDATE #PEDIDO_MAT
SET NUM_ORDEN_ECOMMERCE=SUBSTRING(B.NUM_ORDEN_ECOMMERCE,1,8)
FROM #PEDIDO_MAT A INNER JOIN #TKT_WEB B ON A.COD_TICKET=B.COD_TICKET;

UPDATE #PEDIDO_MAT
SET TipoComprobante=b.TipoComprobante,NumFolio=b.NumFolio
FROM #PEDIDO_MAT A INNER JOIN #Ticket B ON A.COD_TICKET=B.CodTicket_construido;

delete from  dl_sp_dashboards_ecommerce.venta_online_almacen
where cod_ticket in (select distinct cod_ticket from #PEDIDO_MAT );

insert into dl_sp_dashboards_ecommerce.venta_online_almacen
SELECT COD_DIA,COD_TICKET,COD_MATERIAL,MTO_VENTA_NETA,MTO_VENTA_BRUTA,CNT_VENTA,COD_TIENDA,MTO_MARGEN,DESC_MATERIAL,
MARCA,GRUPO_ARTICULO,CATEGORIA,GERENTE,UNIDAD_NEGOCIO,GERENCIA_COMERCIAL,NUM_ORDEN_ECOMMERCE,TipoComprobante,NumFolio,convert_timezone('GMT','GMT+5',getdate()) FECHA_ACTUALIZACION,
TO_DATE(COD_DIA, 'YYYYMMDD')fecha_factura FROM #PEDIDO_MAT
WHERE COD_CANAL=2;

UPDATE dl_sp_dashboards_ecommerce.venta_online_almacen
SET fecha_actualizacion=convert_timezone('GMT','GMT+5',getdate());


end
		   





$$
;
