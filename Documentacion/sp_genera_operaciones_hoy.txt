CREATE OR REPLACE PROCEDURE dl_sp_dashboards_ecommerce.sp_genera_operaciones_hoy()
	LANGUAGE plpgsql
AS $$

declare

var_fecha int;
var_orderid_min_legacy int;
var_orderid_max_legacy int;
var_orderid_min_metroio int;
var_orderid_max_metroio int;
var_orderid_min_wongio int;
var_orderid_max_wongio int;

begin
	

var_fecha 	:=cast(to_char(dateadd(day,-0,convert_timezone('GMT','GMT+5',getdate())),'yyyymmdd') as int);

drop table if exists #data_programados_consolidado_orden;--10 seg
select
from_unixtime_janis(o.date_created)				as fechapedido
,o.seq_id										as op
,o.vtex_id										as vtex_id
,shipping_window_start
,shipping_window_end
into #data_programados_consolidado_orden
from stg_ecommerce_janis_legacy.wms_orders o
left join stg_ecommerce_janis_legacy.wms_stores s on (s.id=o.store)
left join stg_ecommerce_janis_legacy.wms_order_shipping shi on (shi.order_id=o.id)
left join stg_ecommerce_janis_legacy.wms_logistic_carriers c on (c.ref_id=shi.carrier_id)
where to_char(from_unixtime_janis(shi.shipping_window_start),'yyyymmdd')=var_fecha
union all
select
from_unixtime_janis(o.date_created)				as fechapedido
,o.seq_id										as op
,o.vtex_id										as vtex_id
,shipping_window_start
,shipping_window_end
from stg_ecommerce_janis_metroio.wms_orders o
left join stg_ecommerce_janis_metroio.wms_stores s on (s.id=o.store)
left join stg_ecommerce_janis_metroio.wms_order_shipping shi on (shi.order_id=o.id)
left join stg_ecommerce_janis_metroio.wms_logistic_carriers c on (c.ref_id=shi.carrier_id)
where to_char(from_unixtime_janis(shi.shipping_window_start),'yyyymmdd')=var_fecha
union all
select
from_unixtime_janis(o.date_created)				as fechapedido
,o.seq_id										as op
,o.vtex_id										as vtex_id
,shipping_window_start
,shipping_window_end
from stg_ecommerce_janis_wongio.wms_orders o
left join stg_ecommerce_janis_wongio.wms_stores s on (s.id=o.store)
left join stg_ecommerce_janis_wongio.wms_order_shipping shi on (shi.order_id=o.id)
left join stg_ecommerce_janis_wongio.wms_logistic_carriers c on (c.ref_id=shi.carrier_id)
where to_char(from_unixtime_janis(shi.shipping_window_start),'yyyymmdd')=var_fecha;


drop table if exists #cambios_orden;--0 seg
select
vtex_id
,count(distinct shipping_window_start)		as shipping_window_start
,count(distinct shipping_window_end)		as shipping_window_end 
into #cambios_orden
from #data_programados_consolidado_orden
group by
vtex_id;

drop table if exists #pedido_modificado_entrega;--0 seg
select
vtex_id
into #pedido_modificado_entrega
from #cambios_orden
where (shipping_window_start>1 or shipping_window_end>1);


-- Janis Legacy
drop table if exists #data_programados_consolidado_legacy;--7 seg
select
 o.id																																			as order_id
,from_unixtime_janis(o.date_created)																											as fechapedido
,from_unixtime_janis(o.invoice_date)																											as fecha_invoice
,from_unixtime_janis(o.date_picked)																												as fecha_picked
,s.title																																		as title
,o.seq_id																																		as op
,o.vtex_id																																		as vtex_id
,o.picker
,o.status																																		as statusorder
,o.invoice_ammount																																as costinvoice
,upper(website_name)																															as cuenta
,((o.invoice_ammount-o.total_shipping)/1.18)																									as costnet
,from_unixtime_janis(shi.shipping_window_start)																									as wstart
,from_unixtime_janis(shi.shipping_window_end)																									as wend
,from_unixtime_janis(shi.shipped_date_start)																									as shistart
,from_unixtime_janis(shi.shipped_date_end)																										as shiend
,to_char(from_unixtime_janis(shi.shipping_window_start),'HH24:MI')																				as hora_ini
,to_char(from_unixtime_janis(shi.shipping_window_end),'HH24:MI')																				as hora_fin
,'['+to_char(from_unixtime_janis(shi.shipping_window_start),'HH24:MI')+'-'+to_char(from_unixtime_janis(shi.shipping_window_end),'HH24:MI')+']'	as ventana_despacho
,upper(c.type)																																	as tipo
,o.items_qty
,o.items_qty_picked
,o.items_substitute_qty
,o.items_substituted_qty
,o.items_qty_missing
,o.items_qty_original
,shi.id
,convert(varchar(20),'LEGACY')																													as origen
--,shi.*
into #data_programados_consolidado_legacy
from stg_ecommerce_janis_legacy.wms_orders o
left join stg_ecommerce_janis_legacy.wms_stores s on (s.id=o.store)
left join stg_ecommerce_janis_legacy.wms_order_shipping shi on (shi.order_id=o.id)
inner join 	(
			select
			order_id
			, max(id) as id
			from stg_ecommerce_janis_legacy.wms_order_shipping
			group by
			order_id
			) shi_max on (shi_max.order_id=shi.order_id and shi_max.id=shi.id)
left join stg_ecommerce_janis_legacy.wms_logistic_carriers c on (c.ref_id=shi.carrier_id)
where to_char(from_unixtime_janis(shi.shipping_window_start),'yyyymmdd')=var_fecha
and shi_max.id is not null;

-- Janis Metro IO
drop table if exists #data_programados_consolidado_metro;--4 seg
select
 o.id																																			as order_id
,from_unixtime_janis(o.date_created)																											as fechapedido
,from_unixtime_janis(o.invoice_date)																											as fecha_invoice
,from_unixtime_janis(o.date_picked)																												as fecha_picked
,s.title																																		as title
,o.seq_id																																		as op
,o.vtex_id																																		as vtex_id
,o.picker
,o.status																																		as statusorder
,o.invoice_ammount																																as costinvoice
,upper(website_name)																															as cuenta
,((o.invoice_ammount-o.total_shipping)/1.18)																									as costnet
,from_unixtime_janis(shi.shipping_window_start)																									as wstart
,from_unixtime_janis(shi.shipping_window_end)																									as wend
,from_unixtime_janis(shi.shipped_date_start)																									as shistart
,from_unixtime_janis(shi.shipped_date_end)																										as shiend
,to_char(from_unixtime_janis(shi.shipping_window_start),'HH24:MI')																				as hora_ini
,to_char(from_unixtime_janis(shi.shipping_window_end),'HH24:MI')																				as hora_fin
,'['+to_char(from_unixtime_janis(shi.shipping_window_start),'HH24:MI')+'-'+to_char(from_unixtime_janis(shi.shipping_window_end),'HH24:MI')+']'	as ventana_despacho
,upper(c.type)																																	as tipo
,o.items_qty
,o.items_qty_picked
,o.items_substitute_qty
,o.items_substituted_qty
,o.items_qty_missing
,o.items_qty_original
,shi.id
,convert(varchar(20),'METRO_IO')																												as origen
into #data_programados_consolidado_metro
from stg_ecommerce_janis_metroio.wms_orders o
left join stg_ecommerce_janis_metroio.wms_stores s on (s.id=o.store)
left join stg_ecommerce_janis_metroio.wms_order_shipping shi on (shi.order_id=o.id)
inner join 	(
			select
			order_id
			, max(id) as id
			from stg_ecommerce_janis_metroio.wms_order_shipping
			group by
			order_id
			) shi_max on (shi_max.order_id=shi.order_id and shi_max.id=shi.id)
left join stg_ecommerce_janis_metroio.wms_logistic_carriers c on (c.ref_id=shi.carrier_id)
where to_char(from_unixtime_janis(shi.shipping_window_start),'yyyymmdd')=var_fecha
and shi_max.id is not null;

-- Janis Wong IO
drop table if exists #data_programados_consolidado_wong;--4 seg
select
 o.id																																			as order_id
,from_unixtime_janis(o.date_created)																											as fechapedido
,from_unixtime_janis(o.invoice_date)																											as fecha_invoice
,from_unixtime_janis(o.date_picked)																												as fecha_picked
,s.title																																		as title
,o.seq_id																																		as op
,o.vtex_id																																		as vtex_id
,o.picker
,o.status																																		as statusorder
,o.invoice_ammount																																as costinvoice
,upper(website_name)																															as cuenta
,((o.invoice_ammount-o.total_shipping)/1.18)																									as costnet
,from_unixtime_janis(shi.shipping_window_start)																									as wstart
,from_unixtime_janis(shi.shipping_window_end)																									as wend
,from_unixtime_janis(shi.shipped_date_start)																									as shistart
,from_unixtime_janis(shi.shipped_date_end)																										as shiend
,to_char(from_unixtime_janis(shi.shipping_window_start),'HH24:MI')																				as hora_ini
,to_char(from_unixtime_janis(shi.shipping_window_end),'HH24:MI')																				as hora_fin
,'['+to_char(from_unixtime_janis(shi.shipping_window_start),'HH24:MI')+'-'+to_char(from_unixtime_janis(shi.shipping_window_end),'HH24:MI')+']'	as ventana_despacho
,upper(c.type)																																	as tipo
,o.items_qty
,o.items_qty_picked
,o.items_substitute_qty
,o.items_substituted_qty
,o.items_qty_missing
,o.items_qty_original
,shi.id
,convert(varchar(20),'WONG_IO')																													as origen
into #data_programados_consolidado_wong
from stg_ecommerce_janis_wongio.wms_orders o
left join stg_ecommerce_janis_wongio.wms_stores s on (s.id=o.store)
left join stg_ecommerce_janis_wongio.wms_order_shipping shi on (shi.order_id=o.id)
inner join 	(
			select
			order_id
			, max(id) as id
			from stg_ecommerce_janis_wongio.wms_order_shipping
			group by
			order_id
			) shi_max on (shi_max.order_id=shi.order_id and shi_max.id=shi.id)
left join stg_ecommerce_janis_wongio.wms_logistic_carriers c on (c.ref_id=shi.carrier_id)
where to_char(from_unixtime_janis(shi.shipping_window_start),'yyyymmdd')=var_fecha
--and o.id between var_orderid_min_wongio and var_orderid_max_wongio
and shi_max.id is not null;

drop table if exists #data_programados_consolidado;--0 seg
select *
into #data_programados_consolidado
from #data_programados_consolidado_legacy
union all
select *
from #data_programados_consolidado_metro
union all
select *
from #data_programados_consolidado_wong;

alter table #data_programados_consolidado add constraint vtex_id unique (vtex_id);
alter table #data_programados_consolidado add descestado varchar(100);
alter table #data_programados_consolidado add constraint order_id unique (order_id);
alter table #data_programados_consolidado add constraint origen unique (origen);
alter table #data_programados_consolidado_legacy add constraint order_id_legacy unique (order_id);
alter table #data_programados_consolidado_metro add constraint order_id_metro unique (order_id);
alter table #data_programados_consolidado_wong add constraint order_id_wong unique (order_id);



--actualizar estados
update #data_programados_consolidado set--0 seg
descestado	=case statusorder	when 5 then 'PENDIENTE DE AUDITORÍA'
								when 1 then 'NUEVA ORDEN'
								when 6 then 'AUDITORÍA RECHAZADA'
								when 9 then 'PENDIENTE DE PICKING EN EL SELLER'
								when 10 then 'PENDIENTE DE PICKING'
								when 11 then 'PENDIENTE DE REPICKING'
								when 20 then 'EN PICKING'
								when 21 then 'PENDIENTE DE FINALIZAR PICKING'
								when 22 then 'FINALIZANDO PICKING'
								when 23 then 'PREPICKING'
								when 25 then 'PREPICKING'
								when 30 then 'PICKEADO POR FACTURAR'
								when 35 then 'PRECANCELADA'
								when 40 then 'FACTURADO'
								when 50 then 'A RECUPERAR'
								when 60 then 'LISTO PARA ENVIAR'
								when 65 then 'LISTO PARA RETIRAR'
								when 66 then 'ESPERANDO POR LOGISTICA EXTERNA'
								when 70 then 'DESPACHADO'
								when 80 then 'A REPROGRAMAR ENTREGA'
								when 90 then 'ENTREGADO'
								when 95 then 'LIQUIDADO'
								when 100 then 'ANULADO'
								when 105 then 'ANULACIÓN AUTOMÁTICA PENDIENTE'
								when 110 then 'ANULACIÓN CONFIRMADA'
								when 120 then 'CANCELACIÓN PENDIENTE'
								when 125 then 'CANCELADO'
								when 130 then 'DEVOLUCIÓN PARCIAL'
								when 135 then 'DEVOLUCIÓN TOTAL'
								when 33 then 'PROCESANDO DESCUENTOS'
								else null end;

--Nuevo
drop table if exists #eco_oms_general;
select *
into #eco_oms_general
from  stg_ecommerce_vtex.eco_oms
WHERE creationdate>=to_date(getdate()-20,'YYYY-MM-DD');
														
--hora de fecha real de eco_oms
drop table if exists #eco_oms;
select distinct
a.ordernum						as vtex_id
,a.shipped_date_start
,a.prime_order
,a.clientdocument				as dni
into #eco_oms
from #eco_oms_general a inner join #data_programados_consolidado b on (a.ordernum=b.vtex_id)
and a.quantity_sku<>0;

--pedidos creados hoy
drop table if exists #eco_oms_1;
select distinct
a.ordernum						as vtex_id
,a.prime_order
,a.statusorder
,a.clientdocument				as dni
,a.creationdate					as fecha_creacion
,a.paymentsystemname			as medio_pago
into #eco_oms_1
from #eco_oms_general a 
where to_char(a.creationdate,'yyyymmdd') = var_fecha
and a.quantity_sku<>0;


update #data_programados_consolidado set--0 seg
shistart	=b.shipped_date_start
,shiend		=b.shipped_date_start
from #data_programados_consolidado a
inner join #eco_oms b on (a.vtex_id=b.vtex_id)
where (a.shistart is null or a.shistart<='1970-01-01 00:00');

--actualizar con fecha de cambio de estado
drop table if exists #data_hora_despacho_consolidado;--9 seg
select
o.id
,o.vtex_id
,o.statusorder
,os.old_status
,os.new_status
,from_unixtime_janis(os.date_created)		as osfechadate
into #data_hora_despacho_consolidado
from  #data_programados_consolidado_legacy o
left join stg_ecommerce_janis_legacy.wms_order_status_changes os on (o.order_id=os.order_id)
union all
select
o.id
,o.vtex_id
,o.statusorder
,os.old_status
,os.new_status
,from_unixtime_janis(os.date_created)		as osfechadate
from  #data_programados_consolidado_metro o
left join stg_ecommerce_janis_metroio.wms_order_status_changes os on (o.order_id=os.order_id)
union all
select
o.id
,o.vtex_id
,o.statusorder
,os.old_status
,os.new_status
,from_unixtime_janis(os.date_created)		as osfechadate
from  #data_programados_consolidado_wong o
left join stg_ecommerce_janis_wongio.wms_order_status_changes os on (o.order_id=os.order_id);

drop table if exists #pedido_hora_estado_entregado;--0 seg
select
vtex_id
,new_status
,osfechadate fecha_hora_estado
into #pedido_hora_estado_entregado
from #data_hora_despacho_consolidado
where new_status=90
group by
vtex_id
,new_status
,osfechadate;

update #data_programados_consolidado set--0 seg
shistart	=fecha_hora_estado
,shiend		=fecha_hora_estado
from #data_programados_consolidado a
inner join #pedido_hora_estado_entregado b on (a.vtex_id=b.vtex_id)
where (a.shistart<='1970-01-01 00:00:00.0000000' or a.shistart is null);

--tipo de despacho
drop table if exists #data_adic_programado;--0 seg
select distinct
a.*
, case  when d.id_cadena='02' then 'METRO'
		when d.id_cadena='01' then 'WONG' end 																																										as cadena
,to_char(a.wstart,'yyyymmdd')/100 																																													as mes_programado
,substring(a.title,1,4)																																																as cod_tienda
,substring(a.title,7,40)																																															as desc_tienda
,to_char(a.fechapedido,'yyyymmdd')																																													as fecha_pedido
,to_char(a.wstart,'yyyymmdd')																																														as fecha_programado
,to_char(a.shistart,'yyyymmdd')																																														as fecha_entregado 
,to_char(a.fecha_picked,'yyyymmdd')																																													as fechapicked
,convert(varchar(25),a.wstart) 																																														as fecha_programada_ini
,convert(varchar(25),a.wend) 																																														as fecha_programada_fin
,convert(varchar(25),a.shistart) 																																													as fecha_entrega_ini
,convert(varchar(25),a.shiend) 																																														as fecha_entrega_fin
,case 	when a.items_qty_picked>a.items_qty then a.items_qty
		else a.items_qty_picked end 																																												as qpickeada
,case when a.tipo='ENTREGA A DOMICILIO' or tipo='DESPACHO DESDE TIENDA' or a.tipo='DESPACHO A DOMICILIO' 
or a.tipo like '%ENTREGA A DOMICILIO NOCTURNA%'
       or a.tipo='DESPACHADO POR PROVEEDOR' then 'ENTREGA A DOMICILIO' 
   		when a.tipo like '%RETIRO%' then 'RETIRO EN TIENDA'
   		when a.tipo like '%WONG AL AUTO%' then 'WONG AL AUTO'
   		when a.tipo like '%DELIVERY DOM%' then 'DELIVERY POLLO BRASA'
   		when a.tipo like '%EXPRESS%' then 'EXPRESS'
	   	else a.tipo end																																															as tipo_despacho
,b.desc_estado_orden_online 
,upper(b.desc_estado_orden_online_agrupado) 																																										as desc_estado_orden_online_agrupado 
,convert_timezone('GMT','GMT+5',getdate()) 																																																		as hora_consulta
into #data_adic_programado
from #data_programados_consolidado a
left join dl_sp_maestras.dim_estado_orden_online b on (a.statusorder=b.id_estado_orden_online)
left join dl_sp_maestras.dim_tienda c on (substring(a.title,1,4)=c.id_tienda)
left join dl_sp_maestras.dim_cadena d on (c.sk_cadena=d.sk_cadena);
	
update #data_adic_programado set--0 seg
cadena	='WONG'
where cod_tienda='TH12';

update #data_adic_programado set--0 seg
cod_tienda	='W101'
where cod_tienda='TH12';

drop table if exists #data_adic_programado_orden;--0 seg
select distinct
*
,row_number()over(partition by vtex_id order by id desc)		as orden
into #data_adic_programado_orden
from #data_adic_programado;

delete--0 seg
from #data_adic_programado_orden
where orden not in (1);

delete--0 seg
from #data_adic_programado_orden
where desc_estado_orden_online_agrupado in ('ANULADO Y DEVOLUCIONES',null);

alter table #data_adic_programado_orden add q_solicitada numeric(18,2);
alter table #data_adic_programado_orden add q_pickeada numeric(18,2);

update #data_adic_programado_orden set--0 seg
q_solicitada	=0
,q_pickeada		=0
where descestado in 
('PICKEADO POR FACTURAR','PROCESANDO DESCUENTOS','PRE CANCELADO','FACTURADO','PENDIENTE SAC','LISTO PARA ENVIAR',
'LISTO PARA RETIRAR','DESPACHADO','A REPROGRAMAR ENTREGA','ENTREGADO','LIQUIDADO');

------------------------------------------------------------------------------------------------------------------
---------------semaforo----------------------------

alter table #data_adic_programado_orden add semaforo int;
alter table #data_adic_programado_orden add dif_hconsulta_hfinal_ventana int;

update #data_adic_programado_orden set--0 seg
semaforo =0
where descestado in ('PENDIENTE DE AUDITORIA','AUDITORIA RECHAZADA','PENDIENTE DE PICKING','EN PICKING','PREPICKING');

update #data_adic_programado_orden set--0 seg
dif_hconsulta_hfinal_ventana	=datediff(minute,wstart, hora_consulta)
where descestado in ('PENDIENTE DE AUDITORIA','AUDITORIA RECHAZADA','PENDIENTE DE PICKING','EN PICKING','PREPICKING');

--rojo
update #data_adic_programado_orden set--0 seg
semaforo	=1
where dif_hconsulta_hfinal_ventana>=-15
and descestado in ('PENDIENTE DE AUDITORIA','AUDITORIA RECHAZADA','PENDIENTE DE PICKING','EN PICKING','PREPICKING');

--ambar
update #data_adic_programado_orden set--0 seg
semaforo	=2
where dif_hconsulta_hfinal_ventana<-15
and dif_hconsulta_hfinal_ventana>=-30
and descestado in ('PENDIENTE DE AUDITORIA','AUDITORIA RECHAZADA','PENDIENTE DE PICKING','EN PICKING','PREPICKING');

--verde
update #data_adic_programado_orden set--0 seg
semaforo	=3
where dif_hconsulta_hfinal_ventana<-30
and descestado in ('PENDIENTE DE AUDITORIA','AUDITORIA RECHAZADA','PENDIENTE DE PICKING','EN PICKING','PREPICKING');

--LISTO PARA ENVIO
drop table if exists #PEDIDO_HORA_ENVIAR;
SELECT VTEX_ID,NEW_STATUS,max(OSFECHADATE) FECHA_HORA_ESTADO 
INTO #PEDIDO_HORA_ENVIAR FROM #data_hora_despacho_consolidado
WHERE NEW_STATUS=60
group by VTEX_ID,NEW_STATUS;

--LISTO PARA FACTURADO
drop table if exists #PEDIDO_HORA_FACTURADO;
SELECT VTEX_ID,NEW_STATUS,max(OSFECHADATE) FECHA_HORA_ESTADO

--regional 10% de tolerancia
drop table if exists #pedidos_unidades_detalles_10;
select distinct
vtex_id
,ref_id
,sustituto
,measurement_unit 																																																												as un
,sum(quantity*unit_multiplier) 																																																									as unid_solicitadas
,sum(case measurement_unit 	when 'KG' then unit_pickeadas/1000
							when 'UN' then quantity_picked*unit_multiplier end) 																																												as unid_pickeadas_con_susti
,case 	when sum(case	when measurement_unit='KG' and sustituto='NO' then unit_pickeadas/1000
						when measurement_unit='UN' and sustituto='NO' then quantity_picked*unit_multiplier end)>sum(quantity*unit_multiplier) then sum(quantity*unit_multiplier)
		when sum(case 	when measurement_unit='KG' and sustituto='NO' then unit_pickeadas/1000
						when measurement_unit='UN' then quantity_picked*unit_multiplier end)<=sum(quantity*unit_multiplier)	then sum(case 	when measurement_unit='KG' and sustituto='NO' then unit_pickeadas/1000
																																			when measurement_unit='UN' then quantity_picked*unit_multiplier end)
		else sum(case 	when measurement_unit='KG' then unit_pickeadas/1000
						when measurement_unit='UN' then quantity_picked*unit_multiplier end) end 																																								as unid_pickeadas_con_susti_menor_100
,case	when sum(case 	when measurement_unit='KG' and sustituto='NO' then unit_pickeadas/1000 
						when measurement_unit='UN' and sustituto='NO' then quantity_picked*unit_multiplier end)>sum(quantity*unit_multiplier) then sum(quantity*unit_multiplier)
		when sum(case 	when measurement_unit='KG' and sustituto='NO' then unit_pickeadas/1000 
						when measurement_unit='UN' and  sustituto='NO' then quantity_picked*unit_multiplier end)<=sum(quantity*unit_multiplier) then sum((case 	when measurement_unit='KG' and sustituto='NO' then unit_pickeadas/1000
																																								when measurement_unit='UN' and sustituto='NO' then quantity_picked*unit_multiplier end))
		else 0 end 																																																												as unid_pickeadas_sin_susti_menor_100
,sum(case	when measurement_unit='KG' and sustituto='SI' then unit_pickeadas/1000
			when measurement_unit='UN' and sustituto='SI' then quantity_picked*unit_multiplier
			else 0 end) 																																																										as unid_susti
,sum(quantity*unit_multiplier)																																																									as unid_pickeadas_con_susti_final
,sum(quantity*unit_multiplier)																																																									as unid_pickeadas_sin_susti_final 
into #pedidos_unidades_detalles_10
from #data_adic_programado_mat_gerencia
group by
vtex_id
,ref_id
,sustituto
,measurement_unit;

update #pedidos_unidades_detalles_10 set
unid_pickeadas_con_susti_final		=0
,unid_pickeadas_sin_susti_final		=0;

update #pedidos_unidades_detalles_10 set
unid_pickeadas_con_susti_final		=case 	when sustituto='NO' and unid_pickeadas_con_susti_menor_100/nullif(cast(unid_solicitadas as numeric(14,2)),0)>0.9 then unid_solicitadas 
											else 0 end
,unid_pickeadas_sin_susti_final		=case 	when sustituto='NO' and unid_pickeadas_sin_susti_menor_100/nullif(cast(unid_solicitadas as numeric(14,2)),0)>0.9 then unid_solicitadas 
											else 0 end
where un='KG'
and sustituto='NO';

update #pedidos_unidades_detalles_10 set
unid_pickeadas_con_susti_final		=unid_pickeadas_con_susti_menor_100
,unid_pickeadas_sin_susti_final		=unid_pickeadas_sin_susti_menor_100
where un='KG'
and sustituto='SI';

update #pedidos_unidades_detalles_10 set
unid_pickeadas_con_susti_final		=unid_pickeadas_con_susti_menor_100
,unid_pickeadas_sin_susti_final		=unid_pickeadas_sin_susti_menor_100
where un='UN';

drop table if exists #pedidos_unidades_10;
select distinct
vtex_id
,sum(unid_solicitadas)							as unid_solicitadas
,sum(unid_pickeadas_con_susti_final)			as unid_pickeadas_con_susti
,sum(unid_pickeadas_sin_susti_final)			as unid_pickeadas_sin_susti
,sum(unid_susti)								as unid_susti
into #pedidos_unidades_10
from #pedidos_unidades_detalles_10
group by
vtex_id;

update #pedidos_unidades_10 set

									else unid_pickeadas_con_susti end
