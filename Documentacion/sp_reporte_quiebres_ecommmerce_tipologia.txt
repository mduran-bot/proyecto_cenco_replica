-- DROP PROCEDURE dl_sp_dashboards_ecommerce.sp_reporte_quiebres_ecommmerce_tipologia();

CREATE OR REPLACE PROCEDURE dl_sp_dashboards_ecommerce.sp_reporte_quiebres_ecommmerce_tipologia()
	LANGUAGE plpgsql
AS $$
	
	
	
	
	
	

DECLARE
    fecha varchar(8);               -- fecha base (ayer)
    fecha_actual varchar(8);        -- fecha de proceso (hoy)
    ult3dias varchar(8);            -- 3 días atrás
    ult30dias varchar(8);           -- 30 días atrás
    fechastockiniciodia varchar(8); -- día anterior
    ultimos10dias varchar(8);       -- 10 días atrás
    futuros10dias varchar(8);       -- 10 días adelante
    i_gx VARCHAR(8); -- variable temporal para el proceso while
    i  VARCHAR(8); -- variable temporal para el proceso while

BEGIN

    -- Si fecha_in es NULL o vacío, asigna la fecha anterior
  --  IF fecha_in IS NULL OR fecha_in = '' THEN
        fecha := to_char(current_date - interval '1 day', 'yyyymmdd');
 --  ELSE
   --     fecha := fecha_in;
  --  END IF;

    fecha_actual:= to_char(current_date, 'yyyymmdd');


    -- calcular las otras fechas reutilizando la variable fecha
    ult3dias := to_char(to_date(fecha, 'yyyymmdd') - interval '2 days', 'yyyymmdd');
    ult30dias := to_char(to_date(fecha, 'yyyymmdd') - interval '29 days', 'yyyymmdd');
    fechastockiniciodia := to_char(to_date(fecha, 'yyyymmdd') - interval '1 day', 'yyyymmdd');
    ultimos10dias := to_char(to_date(fecha, 'yyyymmdd') - interval '9 days', 'yyyymmdd');
    futuros10dias := to_char(to_date(fecha, 'yyyymmdd') + interval '9 days', 'yyyymmdd');
    i_gx := fecha;
    i  := fecha; 

    --------------------------------------------------------MAESTRO DE PICKING ECOMMERCE --------------------------------------------------------------------

    CREATE TEMPORARY TABLE pedidos AS
    SELECT *
    FROM dl_sp_dashboards_ecommerce.tbl_ods_picking_ecommerce
    WHERE cod_fecha_picking = fecha;


    --------------------------------------------------------MAESTRO DE MATERIALES --------------------------------------------------------------------

    CREATE TEMPORARY TABLE materiales AS
    SELECT
        DISTINCT id_material AS cod_material,
        id_ean_principal,
        desc_material,
        id_categoria,
        desc_categoria,
        desc_grupo_articulo,
        id_grupo_articulo,
        desc_sub_categoria,
        desc_gerente,
        p.desc_jefe_categoria,
        e.desc_gerencia_comercial,
        a.sk_proveedor,
        CASE
            WHEN n.id_proveedor = '0000010000' THEN 'POR IDENTIFICAR'
            WHEN n.id_proveedor < '0000010000' THEN 'EXT'
            ELSE 'NAC'
        END AS origen_proveedor,
        a.sk_tipo_set_venta,
        id_tipo_set_venta,
        desc_tipo_set_venta,
        a.sk_clase_surtido,
        id_clase_surtido,
        desc_clase_surtido,
        a.sk_tipo_surtido_incluido,
        id_tipo_surtido_incluido,
        q.desc_tipo_surtido_incluido,
        sk_material
    FROM
        dl_sp_capa_semantica.dw_cencofcic_dim_material a
        LEFT JOIN dl_sp_maestras.dim_grupo_articulo b ON a.sk_grupo_articulo = b.sk_grupo_articulo
        LEFT JOIN dl_sp_maestras.dim_categoria c ON b.sk_categoria = c.sk_categoria
        LEFT JOIN dl_sp_maestras.dim_gerente d ON b.sk_gerente = d.sk_gerente
        LEFT JOIN dl_sp_maestras.dim_gerencia_comercial  e ON b.sk_gerencia_comercial = e.sk_gerencia_comercial
        LEFT JOIN dl_sp_capa_semantica.dw_cencofcic_dim_marca m ON a.sk_marca = m.sk_marca
        LEFT JOIN dl_sp_capa_semantica.dw_cencofcic_dim_proveedor n ON a.sk_proveedor = n.sk_proveedor
        LEFT JOIN dl_sp_maestras.dim_sub_categoria o ON b.sk_sub_categoria = o.sk_sub_categoria
        LEFT JOIN dl_sp_maestras.dim_jefe_categoria p ON b.sk_jefe_categoria = p.sk_jefe_categoria
        LEFT JOIN dl_sp_maestras.dim_tipo_surtido_incluido q ON a.sk_tipo_surtido_incluido = q.sk_tipo_surtido_incluido
        LEFT JOIN dl_sp_maestras.dim_tipo_set_venta r ON a.sk_tipo_set_venta = r.sk_tipo_set_venta
        LEFT JOIN dl_sp_maestras.dim_clase_surtido s ON a.sk_clase_surtido = s.sk_clase_surtido
    WHERE
        trim(id_grupo_articulo) <> '00000000';


--------------------------------------------------------MAESTRO DE MATERIALES - TIENDA CATALOGADOS--------------------------------------------------------------------


CREATE TEMPORARY TABLE material_tienda_activos (
    cod_material VARCHAR(20),
    cod_tienda VARCHAR(4)
);

-- Bloque 1: Verifica si `cod_dia = fecha` no tiene registros y realiza la inserción correspondiente.
INSERT INTO material_tienda_activos
SELECT
    cod_material,
    cod_tienda
FROM per_super_pd_stg.trx_status_mat
WHERE cod_dia = fecha_actual
AND cod_estado_final IN ('AC', 'BL') --CAMBIO CLOUD NO CONSIDERAR BLOQUEO VENTA Y BLOQUEO ECOMMERCE
AND NOT EXISTS (
    SELECT 1
    FROM per_super_pd_stg.trx_status_mat
    WHERE cod_dia = fecha
);

-- Bloque 2: Si `cod_dia = fecha` tiene registros, realiza la otra inserción.
INSERT INTO material_tienda_activos
SELECT
    cod_material,
    cod_tienda
FROM per_super_pd_stg.trx_status_mat
WHERE cod_dia = fecha
AND cod_estado_final IN ('AC', 'BL'); --CAMBIO CLOUD NO CONSIDERAR BLOQUEO VENTA Y BLOQUEO ECOMMERCE


-- MAESTRO DE MATERIALES - TIENDA CATALOGADOS (GREYSTORE)--------------------------------------------------------------------



CREATE TEMPORARY TABLE material_tienda_activos_greystore_exclusivos (
    cod_material VARCHAR(20),
    cod_tienda VARCHAR(4)
);

-- Si no hay datos para la fecha proporcionada, insertar la información
-- Bloque 1: Verifica si `cod_dia = fecha` no tiene registros y realiza la inserción correspondiente.
INSERT INTO material_tienda_activos_greystore_exclusivos
SELECT
    a.cod_material,
    a.cod_tienda
FROM per_super_pd_stg.trx_status_mat a
INNER JOIN dl_sp_maestras.dim_tienda b
    ON a.cod_tienda = b.id_tienda
WHERE cod_dia = fecha_actual
AND b.sk_formato = 9 
AND NOT EXISTS (
    SELECT 1
    FROM per_super_pd_stg.trx_status_mat
    WHERE cod_dia = fecha
);

-- Bloque 2: Si `cod_dia = fecha` tiene registros, realiza la otra inserción.
INSERT INTO material_tienda_activos_greystore_exclusivos
SELECT
    a.cod_material,
    a.cod_tienda
FROM per_super_pd_stg.trx_status_mat a
INNER JOIN dl_sp_maestras.dim_tienda b
    ON a.cod_tienda = b.id_tienda
WHERE cod_dia = fecha
AND b.sk_formato = 9 ;  -- GREYSTORE


-- Crear una tabla temporal para los resultados combinados

CREATE TEMPORARY TABLE material_tienda_activos_gx (
    cod_material VARCHAR(20),
    cod_tienda VARCHAR(4)
);

-- Insertar datos en material_tienda_activos_gx
INSERT INTO material_tienda_activos_gx
SELECT DISTINCT cod_material, 'GR01' AS cod_tienda
FROM material_tienda_activos
WHERE cod_tienda = 'H003'

UNION

SELECT DISTINCT cod_material, 'GR01' AS cod_tienda
FROM material_tienda_activos_greystore_exclusivos
WHERE cod_tienda = 'GW01'

UNION ALL

SELECT DISTINCT cod_material, 'GR03' AS cod_tienda
FROM material_tienda_activos
WHERE cod_tienda = 'S003'

UNION

SELECT DISTINCT cod_material, 'GR03' AS cod_tienda
FROM material_tienda_activos_greystore_exclusivos
WHERE cod_tienda = 'GW03';


-- MAESTRO DE MATERIALES - TIENDA - PEDIDO CON PICKING-------------------------------------------------------------------



-- Insertar los datos en la tabla temporal

CREATE TEMPORARY TABLE pedidos_campos_basicos AS 
SELECT DISTINCT
    store,
    tienda_origen AS store_origen,
    a.status estado,
    cod_fecha_picking,
    cod_fecha_creacion,
    cod_fecha_picking_order,
    vtex_id,
    picker,
    main_category,
    main_refid,
    main_sku,
    main_unit,
    main_requested_units,
    main_requested_unit_spec,
    main_requested_grams,
    main_requested_gram_spec,
    main_picked_units,
    main_picked_unit_spec,
    main_picked_grams,
    main_picked_gram_spec,
    main_difference_units,
    main_difference_unit_spec,
    main_difference_grams,
    main_difference_gram_spec,
    mto_venta_bruta_comprado
FROM pedidos a ;

    --Crear una columna de unidades pickeadas (sin considerar pesables) para tener control de los perecibles
ALTER TABLE pedidos_campos_basicos ADD COLUMN es_pesable INT;
ALTER TABLE pedidos_campos_basicos ADD COLUMN main_picked_units_actualizado INT;
ALTER TABLE pedidos_campos_basicos ADD COLUMN es_quiebre_antes_sustitucion INT;
ALTER TABLE pedidos_campos_basicos ADD COLUMN tipo_quiebre VARCHAR(50);
ALTER TABLE pedidos_campos_basicos ADD COLUMN main_difference_grams_actualizado DECIMAL(10,2);
ALTER TABLE pedidos_campos_basicos ADD COLUMN cod_tienda CHAR(4);
ALTER TABLE pedidos_campos_basicos ADD COLUMN cod_tienda_origen CHAR(4);

-- Actualizar los valores de las nuevas columnas
UPDATE pedidos_campos_basicos
SET cod_tienda = CASE 
    WHEN LEFT(store, 4) IN ('GM01', 'GW01') THEN 'GR01'
    WHEN LEFT(store, 4) IN ('GM03', 'GW03') THEN 'GR03'
    WHEN LEFT(store, 4) = 'TH12' THEN 'W101'
    ELSE LEFT(store, 4)
END,
cod_tienda_origen = CASE
    WHEN LEFT(store_origen, 4) = 'TH12' THEN 'W101'
    ELSE LEFT(store_origen, 4)
END;

UPDATE pedidos_campos_basicos
SET es_pesable = 0, es_quiebre_antes_sustitucion = 0, tipo_quiebre = 'sin quiebre';

UPDATE pedidos_campos_basicos
SET main_picked_units_actualizado = CASE WHEN main_picked_units > main_requested_units THEN main_requested_units ELSE main_picked_units END;

UPDATE pedidos_campos_basicos
SET main_difference_grams_actualizado = CASE WHEN main_picked_grams > main_requested_grams THEN 0 ELSE main_difference_grams END;

UPDATE pedidos_campos_basicos
SET es_pesable = 1
WHERE main_requested_grams > 0;

UPDATE pedidos_campos_basicos
SET es_quiebre_antes_sustitucion = 1 
WHERE main_picked_units_actualizado < main_requested_units
AND es_pesable = 0;

UPDATE pedidos_campos_basicos
SET tipo_quiebre = CASE
    WHEN es_quiebre_antes_sustitucion = 1 AND main_picked_units = 0 THEN 'es quiebre faltante total'
    WHEN es_quiebre_antes_sustitucion = 1 AND main_picked_units < main_requested_units THEN 'es quiebre incompleto'
END
WHERE es_quiebre_antes_sustitucion = 1
AND es_pesable = 0;

UPDATE pedidos_campos_basicos
SET es_quiebre_antes_sustitucion = 1
WHERE (
    CASE
        WHEN main_picked_grams > main_requested_grams THEN main_requested_grams
        ELSE main_picked_grams
    END
) < main_requested_grams
AND es_pesable = 1;

UPDATE pedidos_campos_basicos
SET tipo_quiebre = CASE
    WHEN es_quiebre_antes_sustitucion = 1 AND main_picked_grams = 0 THEN 'es quiebre faltante total'
    WHEN es_quiebre_antes_sustitucion = 1 AND main_picked_grams < main_requested_grams THEN 'es quiebre incompleto'
END
WHERE es_quiebre_antes_sustitucion = 1
AND es_pesable = 1;

--------------------------------------------------------MAESTRO DE MATERIALES - TIENDA - DIA CREACION CON PICKING-------------------------------------------------------------------

CREATE TEMPORARY TABLE maestro_material_tienda_quiebre as 
select
    cod_tienda,
    cod_fecha_picking fecha_picking,
    right(replicate('0', 18) + main_refid, 18) cod_material,
    case
        when es_pesable = 0 then 'no es pesable'
        else 'es pesable'
    end es_pesable,
    tipo_quiebre,
    case
        when es_quiebre_antes_sustitucion = 1 then 'es quiebre antes de sustitucion'
        else 'sin quiebre'
    end es_quiebre_antes_sustitucion,
    COUNT(distinct vtex_id) Trx,
    SUM(mto_venta_bruta_comprado) Monto_Venta_Bruta_Requerido,
    SUM(main_difference_grams_actualizado) Q_gramos_no_pickeado_solo_pesable,
    SUM(main_requested_units) Q_unidades_requeridas,
    SUM(main_picked_units_actualizado) Q_unidades_pickeadas,
    SUM(main_requested_grams) Q_gramos_requerido_solo_pesable
from
    pedidos_campos_basicos a
    left join dl_sp_maestras.dim_estado_orden_online b on a.estado = b.SK_ESTADO_ORDEN_ONLINE
    left join materiales e on right(replicate('0',18)+main_refid,18) = e.COD_MATERIAL
group by
    cod_tienda,
    cod_fecha_picking,
    main_refid,
    case
        when es_pesable = 0 then 'no es pesable'
        else 'es pesable'
    end,
    tipo_quiebre,
    case
        when es_quiebre_antes_sustitucion = 1 then 'es quiebre antes de sustitucion'
        else 'sin quiebre'
    end; -- Crear la tabla maestro con los valores sumados


CREATE TEMPORARY TABLE maestro AS
select
    cod_tienda,
    fecha_picking,
    cod_material,
    es_pesable,
    SUM(Trx) Trx,
    SUM(Monto_Venta_Bruta_Requerido) Monto_Venta_Bruta_Requerido,
    SUM(Q_gramos_no_pickeado_solo_pesable) Q_gramos_no_pickeado_solo_pesable,
    SUM(Q_unidades_requeridas) Q_unidades_requeridas,
    SUM(Q_unidades_pickeadas) Q_unidades_pickeadas,
    SUM(Q_gramos_requerido_solo_pesable) Q_gramos_requerido_solo_pesable
from
    maestro_material_tienda_quiebre
group by
    cod_tienda,
    fecha_picking,
    cod_material,
    es_pesable; -- Crear las tablas auxiliares



CREATE TEMPORARY TABLE aux_quiebre AS
select
    cod_tienda,
    fecha_picking,
    cod_material,
    es_quiebre_antes_sustitucion,
    SUM(Trx) Trx,
    SUM(Monto_Venta_Bruta_Requerido) Monto_Venta_Bruta_Requerido,
    SUM(Q_gramos_no_pickeado_solo_pesable) Q_gramos_no_pickeado_solo_pesable,
    SUM(Q_unidades_requeridas) Q_unidades_requeridas,
    SUM(Q_unidades_pickeadas) Q_unidades_pickeadas,
    SUM(Q_gramos_requerido_solo_pesable) Q_gramos_requerido_solo_pesable
from
    maestro_material_tienda_quiebre
GROUP BY
    cod_tienda,
    fecha_picking,
    cod_material,
    es_quiebre_antes_sustitucion; -- Crear tablas auxiliares para quiebre

CREATE TEMPORARY TABLE aux_quiebre_apertura AS
select
    cod_tienda,
    fecha_picking,
    cod_material,
    tipo_quiebre,
    SUM(Trx) Trx,
    SUM(Monto_Venta_Bruta_Requerido) Monto_Venta_Bruta_Requerido,
    SUM(Q_gramos_no_pickeado_solo_pesable) Q_gramos_no_pickeado_solo_pesable,
    SUM(Q_unidades_requeridas) Q_unidades_requeridas,
    SUM(Q_unidades_pickeadas) Q_unidades_pickeadas,
    SUM(Q_gramos_requerido_solo_pesable) Q_gramos_requerido_solo_pesable 
from
    maestro_material_tienda_quiebre
GROUP BY
    cod_tienda,
    fecha_picking,
    cod_material,
    tipo_quiebre; -- Crear tablas auxiliares para quiebre apertura

alter table
    maestro add Trx_Con_quiebre int;
alter table
    maestro add Monto_Venta_Bruta_Requerido_Con_quiebre numeric(10,2);

    maestro add Monto_Venta_Bruta_Requerido_Con_quiebre_Faltante numeric(10,2);
alter table
    maestro add Q_gramos_no_pickeado_solo_pesable_Con_quiebre_Faltante numeric(10,2);

    maestro add Q_unidades_requeridas_Con_quiebre_Faltante numeric(10,2);
alter table
    maestro add Q_unidades_pickeadas_Con_quiebre_Faltante numeric(10,2);
alter table
    maestro add Q_gramos_requerido_solo_pesable_Con_quiebre_Faltante numeric(10,2);
alter table
    maestro add Trx_Con_quiebre_Incompleto int;
alter table
    maestro add Monto_Venta_Bruta_Requerido_Con_quiebre_Incompleto numeric(10,2);
alter table
    maestro add Q_gramos_no_pickeado_solo_pesable_Con_quiebre_Incompleto numeric(10,2);
alter table
    maestro add Q_unidades_requeridas_Con_quiebre_Incompleto numeric(10,2);
alter table
    maestro add Q_unidades_pickeadas_Con_quiebre_Incompleto numeric(10,2);
alter table
    maestro add Q_gramos_requerido_solo_pesable_Con_quiebre_Incompleto numeric(10,2);
alter table
    maestro add Trx_Sin_quiebre int;
alter table
    maestro add Monto_Venta_Bruta_Requerido_Sin_quiebre numeric(10,2);
alter table
    maestro add Q_gramos_no_pickeado_solo_pesable_Sin_quiebre numeric(10,2);
alter table
    maestro add Q_unidades_requeridas_Sin_quiebre numeric(10,2);
alter table
    maestro add Q_unidades_pickeadas_Sin_quiebre numeric(10,2);
alter table
    maestro add Q_gramos_requerido_solo_pesable_Sin_quiebre numeric(10,2);

UPDATE
    maestro SET Trx_Con_quiebre = 0, Monto_Venta_Bruta_Requerido_Con_quiebre = 0, Q_gramos_no_pickeado_solo_pesable_Con_quiebre = 0,
    Q_unidades_requeridas_Con_quiebre = 0,
    Q_unidades_pickeadas_Con_quiebre = 0,
    Q_gramos_requerido_solo_pesable_Con_quiebre = 0,
    Trx_Sin_quiebre = 0,
    Monto_Venta_Bruta_Requerido_Sin_quiebre = 0,
    Q_gramos_no_pickeado_solo_pesable_Sin_quiebre = 0,
    Q_unidades_requeridas_Sin_quiebre = 0,
    Q_unidades_pickeadas_Sin_quiebre = 0,
    Q_gramos_requerido_solo_pesable_Sin_quiebre = 0,
    Trx_Con_quiebre_incompleto = 0,
    Monto_Venta_Bruta_Requerido_Con_quiebre_incompleto = 0,
    Q_gramos_no_pickeado_solo_pesable_Con_quiebre_incompleto = 0,
    Q_unidades_requeridas_Con_quiebre_incompleto = 0,
    Q_unidades_pickeadas_Con_quiebre_incompleto = 0,
    Q_gramos_requerido_solo_pesable_Con_quiebre_incompleto = 0,
    Trx_Con_quiebre_faltante = 0,
    Monto_Venta_Bruta_Requerido_Con_quiebre_faltante = 0,
    Q_gramos_no_pickeado_solo_pesable_Con_quiebre_faltante = 0,
    Q_unidades_requeridas_Con_quiebre_faltante = 0,
    Q_unidades_pickeadas_Con_quiebre_faltante = 0,
    Q_gramos_requerido_solo_pesable_Con_quiebre_faltante = 0 ;

UPDATE maestro
SET 
    Trx_Con_quiebre = b.Trx,
    Monto_Venta_Bruta_Requerido_Con_quiebre = b.Monto_Venta_Bruta_Requerido,
    Q_gramos_no_pickeado_solo_pesable_Con_quiebre = b.Q_gramos_no_pickeado_solo_pesable,
    Q_unidades_requeridas_Con_quiebre = b.Q_unidades_requeridas,
    Q_unidades_pickeadas_Con_quiebre = b.Q_unidades_pickeadas,
    Q_gramos_requerido_solo_pesable_Con_quiebre = b.Q_gramos_requerido_solo_pesable
FROM aux_quiebre b
WHERE maestro.cod_material = b.cod_material
  AND maestro.COD_TIENDA = b.cod_tienda
  AND maestro.fecha_picking = b.fecha_picking
  AND b.es_quiebre_antes_sustitucion = 'es quiebre antes de sustitucion';


UPDATE maestro
SET 
    Trx_Sin_quiebre = b.Trx,
    Monto_Venta_Bruta_Requerido_Sin_quiebre = b.Monto_Venta_Bruta_Requerido,
    Q_gramos_no_pickeado_solo_pesable_Sin_quiebre = b.Q_gramos_no_pickeado_solo_pesable,
    Q_unidades_requeridas_Sin_quiebre = b.Q_unidades_requeridas,
    Q_unidades_pickeadas_Sin_quiebre = b.Q_unidades_pickeadas,
    Q_gramos_requerido_solo_pesable_Sin_quiebre = b.Q_gramos_requerido_solo_pesable
FROM aux_quiebre b
WHERE maestro.cod_material = b.cod_material
  AND maestro.COD_TIENDA = b.cod_tienda
  AND maestro.fecha_picking = b.fecha_picking
  AND b.es_quiebre_antes_sustitucion = 'sin quiebre';


UPDATE maestro
SET 
    Trx_Con_quiebre_faltante = b.Trx,
    Monto_Venta_Bruta_Requerido_Con_quiebre_faltante = b.Monto_Venta_Bruta_Requerido,
    Q_gramos_no_pickeado_solo_pesable_Con_quiebre_faltante = b.Q_gramos_no_pickeado_solo_pesable,
    Q_unidades_requeridas_Con_quiebre_faltante = b.Q_unidades_requeridas,
    Q_unidades_pickeadas_Con_quiebre_faltante = b.Q_unidades_pickeadas,
    Q_gramos_requerido_solo_pesable_Con_quiebre_faltante = b.Q_gramos_requerido_solo_pesable
FROM aux_quiebre_apertura b
WHERE maestro.cod_material = b.cod_material
  AND maestro.COD_TIENDA = b.cod_tienda
  AND maestro.fecha_picking = b.fecha_picking
  AND b.tipo_quiebre = 'es quiebre faltante total';


UPDATE maestro
SET 
    Trx_Con_quiebre_incompleto = b.Trx,
    Monto_Venta_Bruta_Requerido_Con_quiebre_incompleto = b.Monto_Venta_Bruta_Requerido,
    Q_gramos_no_pickeado_solo_pesable_Con_quiebre_incompleto = b.Q_gramos_no_pickeado_solo_pesable,
    Q_unidades_requeridas_Con_quiebre_incompleto = b.Q_unidades_requeridas,
    Q_unidades_pickeadas_Con_quiebre_incompleto = b.Q_unidades_pickeadas,
    Q_gramos_requerido_solo_pesable_Con_quiebre_incompleto = b.Q_gramos_requerido_solo_pesable
FROM aux_quiebre_apertura b
WHERE maestro.cod_material = b.cod_material
  AND maestro.COD_TIENDA = b.cod_tienda
  AND maestro.fecha_picking = b.fecha_picking
  AND b.tipo_quiebre = 'es quiebre incompleto';


DELETE FROM maestro 
WHERE 
cod_material in (
    SELECT COD_MATERIAL
    FROM materiales

CREATE TEMPORARY TABLE maestro_quiebre_ecommerce_stock_venta AS
SELECT
    A.cod_material,
    A.cod_tienda,
    COALESCE(B.cnt_stock_inicio, 0) AS cnt_stock_inicio_dia_picking,
    COALESCE(B.mto_stock_inicio, 0) AS mto_stock_inicio_dia_picking,
    COALESCE(B.cnt_stock_total_inicio, 0) AS cnt_stock_total_inicio_dia_picking,
    COALESCE(B.mto_stock_total_inicio, 0) AS mto_stock_total_inicio_dia_picking,
    COALESCE(E.cnt_stock_fin, 0) AS cnt_stock_fin_dia_picking,
    COALESCE(E.mto_stock_fin, 0) AS mto_stock_fin_dia_picking,
    COALESCE(E.cnt_stock_total_fin, 0) AS cnt_stock_total_fin_dia_picking,
    COALESCE(E.mto_stock_total_fin, 0) AS mto_stock_total_fin_dia_picking,
    COALESCE(C.cnt_venta, 0) AS cnt_venta,
    COALESCE(C.mto_venta_bruta, 0) AS mto_venta_bruta,
    COALESCE(C.mto_venta_neta, 0) AS mto_venta_neta,
    COALESCE(C.ultfechaventa, 0) AS ultdiaventa,
    DATEDIFF(
        day, 
        TO_DATE(ultfechaventa, 'YYYYMMDD'), 
        TO_DATE(fecha, 'YYYYMMDD')
    )
    AS diassinventa,
    D.id_tipo_set_venta,
    D.desc_tipo_set_venta,
    D.id_clase_surtido,
    D.desc_clase_surtido,
    D.id_tipo_surtido_incluido,
    D.desc_tipo_surtido_incluido,
    D.origen_proveedor
FROM
    combinatorias_material_tienda_quiebre_ecommerce A
    LEFT JOIN stock_inicio_dia_picking B 
        ON A.cod_material = B.cod_material 
        AND A.cod_tienda = B.cod_tienda
    LEFT JOIN stock_fin_dia_picking E 
        ON A.cod_material = E.cod_material 
        AND A.cod_tienda = E.cod_tienda
    LEFT JOIN venta C 
        ON A.cod_material = C.cod_material 
        AND A.cod_tienda = C.cod_tienda
    LEFT JOIN materiales D 
        ON A.cod_material = D.cod_material;

--EXTRAER STOCK DE LOS DIAS DESDE QUE LAS COMBINATORIAS NO TIENEN VENTA
    

    create TEMPORARY table new_temp_gx (
        cod_dia INT,
        cod_material VARCHAR(18),
        cod_tienda VARCHAR(4),
        cnt_stock FLOAT,
        mto_stock FLOAT
    );


    create TEMPORARY table new_temp2_gx (
        cod_dia INT,
        cod_material VARCHAR(18),
        cod_tienda VARCHAR(4),
        cnt_stock FLOAT,
        mto_stock FLOAT
    );

    create TEMPORARY table historiastock_gx (
        cod_dia INT,
        cod_material VARCHAR(18),
        cod_tienda VARCHAR(4),
        cnt_stock FLOAT,
        mto_stock FLOAT
    );


    while i_gx >= ult30dias 
    loop
    
        insert into new_temp_gx
        select
            cast(i_gx as INT  )  cod_dia,
            cod_material,
            cod_tienda,
            0,
            0
        from
            maestro_quiebre_ecommerce_stock_venta 
        where i_gx >= ultdiaventa;

        insert into new_temp2_gx
        select
            v.cod_dia,
            v.cod_material,
            v.cod_tienda,
            sum(v.cnt_stock_tienda) as cnt_stock_ultimo,
            sum(v.mto_stock_tienda) as mto_stock_ultimo
        from
            stock_gr0x_maestro v
        inner join maestro_quiebre_ecommerce_stock_venta s 
            on v.cod_material = s.cod_material 
            and v.cod_tienda = s.cod_tienda 
            and v.cod_dia >= s.ultdiaventa
        where v.cod_dia = i_gx
        group by
            v.cod_dia, v.cod_material, v.cod_tienda
        having sum(v.cnt_stock_tienda) > 0;

        
        insert into historiastock_gx
        select * from new_temp2_gx
        union all
        select t.*
        from new_temp_gx t
        left outer join new_temp2_gx h on t.cod_material = h.cod_material and t.cod_tienda = h.cod_tienda
        where h.cod_material is null and h.cod_tienda is null;

        -- Actualización de la fecha I_GX para la siguiente iteración
        i_gx := TO_CHAR(TO_DATE(i_gx, 'yyyymmdd') - INTERVAL '1 day', 'yyyymmdd'); 

        -- Imprimir la fecha actual para depuración
        raise notice 'Procesando fecha: %', i_gx;

        -- Limpieza de las tablas temporales
        delete from new_temp_gx;
        delete from new_temp2_gx;
    end loop;

CREATE TEMPORARY TABLE cambiodestock_gx (

    cod_tienda VARCHAR(4),
    diaanterioringreso INT,
    diaingreso INT
);

INSERT INTO cambiodestock_gx
SELECT
    cod_material,
    cod_tienda,
    MIN(maximo) AS diaanterioringreso,
    MIN(m) AS diaingreso
FROM (
    SELECT
        A.cod_material,
        A.cod_tienda,
        MAX(A.cod_dia) AS maximo,
        CAST(TO_CHAR(
            TO_DATE(MAX(A.cod_dia), 'YYYYMMDD') + INTERVAL '1 day', 
            'YYYYMMDD'
        ) AS INT ) AS m
    FROM
        historiastock_gx A
    INNER JOIN historiastock_gx B 
        ON A.cod_material = B.cod_material 
        AND A.cod_tienda = B.cod_tienda 
        AND A.cnt_stock < B.cnt_stock
    WHERE
        A.cod_dia < B.cod_dia
    GROUP BY
        A.cod_material,
        A.cod_tienda
) A
GROUP BY
    cod_material,
    cod_tienda;

--------------------------IDENTIFICAR CAMBIO DE STOCK----------------------------------
-- EN LOS DIAS EN DONDE NO SE VENDE EL PRODUCTO SE IDENTIFICA EL DÍA EN DONDE INGRESO MERCADERIA (CAMBIO DE STOCK)


create TEMPORARY table cambiodestock as 

select
    cod_material,
    cod_tienda,
    min(maximo) diaanterioringreso,
    min(m) diaingreso
from
(
    select
        a.cod_material,
        a.cod_tienda,
        max(a.cod_dia) maximo,
        to_char(to_date(max(a.cod_dia)::text, 'yyyymmdd') + interval '1 day', 'yyyymmdd')::integer as m
    from
        historiastock a
    inner join historiastock b 
        on a.cod_material = b.cod_material 
        and a.cod_tienda = b.cod_tienda 
        and a.cnt_stock < b.cnt_stock
    where
        a.cod_dia < b.cod_dia
    group by
        a.cod_material, a.cod_tienda
) a
group by
    cod_material, cod_tienda;

    SUM(cnt_stock_tda) AS cnt_stock_fin,
    SUM(mto_stock_tda) AS mto_stock_fin,
    SUM(cnt_stock) AS cnt_stock_total_fin,
    SUM(mto_stock) AS mto_stock_total_fin
FROM
    per_super_pd_stg.trx_stock_mat a
    INNER JOIN faltante_material_tienda_dia_creacion b ON a.cod_material = b.cod_material AND a.cod_tienda = b.cod_tienda AND a.cod_dia = b.fecha_creacion
GROUP BY
    a.cod_material,
    a.cod_tienda,
    a.cod_dia;


CREATE TEMPORARY TABLE faltante_material_tienda_dia_creacion_resultado AS
SELECT
    a.*,
    COALESCE(cnt_stock_total_inicio, 0) AS cnt_stock_total_inicio,
    CASE
        WHEN COALESCE(cnt_stock_total_inicio, 0) <= 0 THEN 1
        ELSE 0
    END AS id_sistemas
FROM
    faltante_material_tienda_dia_creacion a
    LEFT JOIN stock_inicio_dia_compra b ON a.cod_material = b.cod_material AND a.cod_tienda = b.cod_tienda AND a.fecha_creacion = b.fecha_creacion;


CREATE TEMPORARY TABLE id_sistemas AS
SELECT
    DISTINCT cod_fecha_picking,
    cod_material,
    cod_tienda,
    CASE
        WHEN id_sistemas > 0 THEN 1
        ELSE 0
    END AS id_sistemas
FROM
(
    SELECT
        cod_fecha_picking,
        RIGHT(REPLICATE('0', 18) + a.main_refid, 18) AS cod_material,
        a.cod_tienda,
        SUM(id_sistemas) AS id_sistemas
    FROM
        pedidos_campos_basicos a
        LEFT JOIN faltante_material_tienda_dia_creacion_resultado b ON a.cod_fecha_creacion = b.fecha_creacion AND RIGHT(REPLICATE('0', 18) + a.main_refid, 18) = b.cod_material AND a.cod_tienda_origen = b.cod_tienda
    WHERE
        tipo_quiebre <> 'sin quiebre'
    GROUP BY
        cod_fecha_picking,
        RIGHT(REPLICATE('0', 18) + a.main_refid, 18),
        a.cod_tienda
) a
WHERE
    id_sistemas > 0;
